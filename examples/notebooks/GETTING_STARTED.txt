from app.core.exam_builder import generate_question

# Your snippet from RAG/vector DB
snippet = """
–ì–µ—Å—Ç–∞—Ü–∏–æ–Ω–Ω–∞—è –≥–∏–ø–µ—Ä—Ç–µ–Ω–∑–∏—è ‚Äî —ç—Ç–æ –∞—Ä—Ç–µ—Ä–∏–∞–ª—å–Ω–æ–µ –¥–∞–≤–ª–µ–Ω–∏–µ ‚â•140/90 –º–º —Ä—Ç. —Å—Ç.,
–≤–ø–µ—Ä–≤—ã–µ –≤—ã—è–≤–ª–µ–Ω–Ω–æ–µ –ø–æ—Å–ª–µ 20-–π –Ω–µ–¥–µ–ª–∏ –±–µ—Ä–µ–º–µ–Ω–Ω–æ—Å—Ç–∏.
"""

# Generate question
question = generate_question(
    content=snippet,
    question_type="single_choice",  # or "multiple_choice"
    difficulty="medium",
    language="ru"  # or "en"
)

# Use the question
print(question['stem'])      # Question text
print(question['options'])   # ["Option 1", "Option 2", ...]
print(question['correct'])   # [0] - indices of correct answers

‚ùì TROUBLESHOOTING:
================================================================================

Problem: "OPENAI_API_KEY not set in environment"
Solution:
  1. cp .env.example .env
  2. Edit .env: add OPENAI_API_KEY=sk-your-key
  3. Restart Jupyter kernel or Python session

Problem: Questions in wrong language
Solution:
  Set language="ru" or language="en" explicitly

Problem: ModuleNotFoundError
Solution:
  pip install -r requirements.txt

Problem: Can't find .env file
Solution:
  .env must be in project root: /home/toharhymes/work/itmo/llm_tester/.env
